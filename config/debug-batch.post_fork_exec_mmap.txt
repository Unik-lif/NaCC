# Post-fork+exec mmap/shared-memory suite.
# These commands are slightly heavier than the busybox-only suite.

docker run --security-opt seccomp=unconfined --rm python:3.11-slim python -c "exec(\"import mmap,os,sys\\nbuf=mmap.mmap(-1,4096,flags=mmap.MAP_SHARED)\\nbuf[:5]=b'hello'\\npid=os.fork()\\nif pid==0:\\n    buf[5:11]=b'child!'\\n    os._exit(0)\\nos.waitpid(pid,0)\\nsys.stdout.write(buf[:11].decode()+'\\\\n')\")"
docker run --security-opt seccomp=unconfined --rm python:3.11-slim python -c "exec(\"import mmap,os,tempfile,sys\\nfd,path=tempfile.mkstemp()\\nos.write(fd,b'0'*4096)\\nos.lseek(fd,0,0)\\nbuf=mmap.mmap(fd,4096,flags=mmap.MAP_SHARED,prot=mmap.PROT_READ|mmap.PROT_WRITE)\\npid=os.fork()\\nif pid==0:\\n    buf[:4]=b'data'\\n    buf.flush()\\n    os._exit(0)\\nos.waitpid(pid,0)\\nos.lseek(fd,0,0)\\nsys.stdout.write(os.read(fd,4).decode()+'\\\\n')\\nos.close(fd)\\nos.unlink(path)\")"
docker run --security-opt seccomp=unconfined --rm python:3.11-slim python -c "exec(\"from multiprocessing.shared_memory import SharedMemory\\nimport os,sys\\nshm=SharedMemory(create=True,size=4096)\\npid=os.fork()\\nif pid==0:\\n    child=SharedMemory(name=shm.name)\\n    child.buf[:4]=b'ping'\\n    child.close()\\n    os._exit(0)\\nos.waitpid(pid,0)\\nsys.stdout.write(bytes(shm.buf[:4]).decode()+'\\\\n')\\nshm.close()\\nshm.unlink()\")"
docker run --security-opt seccomp=unconfined --rm python:3.11-slim python -c "exec(\"import mmap,os,tempfile,sys\\nfd,path=tempfile.mkstemp()\\nos.write(fd,b'abcd'*1024)\\nos.lseek(fd,0,0)\\nbuf=mmap.mmap(fd,4096,flags=mmap.MAP_PRIVATE,prot=mmap.PROT_READ|mmap.PROT_WRITE)\\npid=os.fork()\\nif pid==0:\\n    buf[:4]=b'zzzz'\\n    os._exit(0)\\nos.waitpid(pid,0)\\nsys.stdout.write(buf[:4].decode()+'\\\\n')\\nos.close(fd)\\nos.unlink(path)\")"
